server {
  port 80;
  listen 8001;
  host 127.0.0.1;
  server_name test;
  error_page_1 404 /error/404.html;
  error_page_2 400 /error/400.html;
  client_max_body_size 1024;

  location / { 
      root docs/fusion_web;
      autoindex on;
      allow_methods POST GET;
      index index.html;
      return abc/index1.html;
      alias  docs/fusion_web;
    }
  location cgi-bin  {
      root ./;
      cgi_path /usr/bin/python3 /bin/bash;
      cgi_ext .py .sh;
    }
}
